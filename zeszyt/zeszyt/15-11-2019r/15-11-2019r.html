#!/bin/bash
echo "Wybierz komende 1-mkdir, 2-ls, 3-apt install, 4-man, 5-sudo, 6-chmod, 7-touch"
read d
case "$d" in

"1") echo "mkdir - komenda sluzy do utworzenia nowego katalogu, katalog mozemy nazwac dowolna nazwa mozemy uzyc kazdej litery ktora mozemy znalezc na klawiaturze podkreslam, ze literze tak wiec nazwa katalogu to moze byc nawet 'afbn' w katalogu w ktorym sie obecnie znajdujemy, mozemy zmienic piszac wczesniej sciezke dostepu do katalogu w ktorym chcemy utworzyc nowy katalog" ;;

"2") echo "ls - komenda sluzy do pokazania zawartosci calego katalogu, po dodaniu odpowiednich parametrow mozemy sprecyzowac np. czy chcemy sprawdzic jakie pliki znajduja sie na koncu (tail) albo na poczatku (head) oraz wiele innych. np. parametr -l umozliwa zobaczenie wszystkich plikow jest tez duzo roznych innych parametrow a wymienienie ich wszystkich zajelo by mi cale wieki dlatego napisze po prostu, ze aby dowiedzeic sie jeszcze wiecej o tej komendzie nalezy napisac ls --help" ;;

"3") echo "apt install - komedna sluzy do instalowania nowych bilbiotek, skryptow jesli ich nie mamy a chcemy z nich korzystac. Mozna rownierz zainstalowac aplikacje. Komenda ta wymaga uprawnien roota, dlatego aby jej uzyc trzeba byc zalogowany jako root, albo wpisac wczesniej sudo. Jednym z zastosowan moze byc na przyklad sudo apt install sudoku podana komenda instaluje nam sudoku w terminalu ktorego bedziemy mogli uzywac jak bedzie sie nam nudzilo a nie bedziemy mieli internetu, wazne aby je zaisntalowa jak jeszcze bedziemy mieli dostep do internetu";;

"4") echo "man - komeda sluzy do sprawdzenia co robi inna komenda, wiec jesli chcemy sprawdzic co robi jakas komedna, to zamiast szukac i myslec czy jest tutaj w tym poradniku, mozemy po prostu napisac man i ta komende i wtedy pokaza sie nam wszystkie parametry jakie mozemy do niej wpisac oraz wszystkie jej znaczenia jednym ciekawyszych polecen jakie mozemy wpisac przy uzyciu komendy man to jest man man i to pokazuje caly slownik w ktorym sa wszystkie dostepne komendy" ;;

"5") echo "sudo - piszemy do przed komenda jesli chcemy uzyskac dodatkowe uprawnienia przed uzyciem jakiejs komendy nastepnie trzeba podac haslo do sudo. komenda przydaje sie, jesli chcemy cos zaisntalowac przy pomocy apt install i nie jestesmy na koncie roota, to musimy wczesniej umiescic polecenie sudo aby polecenie zostalo wykonane poprawnie. wazne jest to, ze to daje nam uprawnienia tylko i wylacznie podczas uzywania sudo, jak juz sie skonczy wykonywac dane polecenie to nie bedziemy mieli dostepu do plikow do ktorych potrzebne sa dodatkowe upranienia" ;;

"6") echo "chmod - polecenie sluzy do nadawania uprawnien plikom po wpisaniu danej komendy oraz nazwy pliku trzeba wpisac 3 cyfry pisane razem, pierwsza z nich dotyczy wlasciciela pliku - nas samych - a 7 oznacza pelne uprawnienia do czytania, edytowania oraz uruchamiania skryptu, wiec jesli chcemy tylko my miec dostep do danego pliku piszemy - chmod 700 nazwapliku.rozszerzenie. jesli natomiast chcemy dac kazdemu uzytkownikowi na tym komputerze pelny dostep do tego pliku trzeba wpisac chmod 777 nazwapliku.rozszerzenie, a jesli nie chcemy dac zadncyh uprawnien to trzeba dac trzy zera" ;;

"7") echo "touch - dzieki komendzie touch jestesmy w stanie stworzyc dowolny plik o dowonlej nazwie i dowolnym rozszerzeniu w katalogu w ktorym sie obecnie znajdujemy, jesli natomiast chcemy stworzyc dany plik w innym katalogu nalezy podac do niego sciezke dostepu, nalezy pamietac aby w nazwie po kropce napisac rozszerzenie pliku, w innym przypadku nasz plik bedzie bez rozszerzenia. po raz kolejny przypomne ze nalezy podac rozszerzenie gdyz w przypadku jego nie podania plik nie bedzie sie uruchamiac bo nie bedzie wiedzial w jakim programie ma sie otworzyc" ;;

*) echo "wybierz poprawny numer dla odpowedniej komendy";;

esac
